Instead, the components shown below are provided globally through the registry in `client/src/components/mdx/index.tsx`, so they are available on every page without an import.

### Interactive plot (sliders, checkboxes, selects)

<InteractivePlot
  controls={[
    { id: "freq", type: "range", label: "Frequency", min: 1, max: 10, step: 1, default: 3 },
    { id: "amp", type: "range", label: "Amplitude", min: 10, max: 140, step: 10, default: 80 },
    { id: "fill", type: "checkbox", label: "Fill area", default: false }
  ]}
  draw={`
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.strokeStyle = "#2f81f7";
    ctx.lineWidth = 2;
    ctx.beginPath();
    for (let x = 0; x <= canvas.width; x++) {
      const y = canvas.height / 2 - params.amp * Math.sin((x / canvas.width) * params.freq * 2 * Math.PI);
      if (x === 0) { ctx.moveTo(x, y); } else { ctx.lineTo(x, y); }
    }
    ctx.stroke();
    if (params.fill) {
      ctx.lineTo(canvas.width, canvas.height / 2);
      ctx.lineTo(0, canvas.height / 2);
      ctx.closePath();
      ctx.fillStyle = "rgba(47, 129, 247, 0.2)";
      ctx.fill();
    }
  `}
  width={600}
  height={300}
/>

### Canvas animation (no controls)

<CanvasWithJs code={`
  let x = 60, y = 60, vx = 3.2, vy = 2.4, r = 20;
  function frame() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    x += vx; y += vy;
    if (x < r || x > canvas.width - r) vx = -vx;
    if (y < r || y > canvas.height - r) vy = -vy;
    ctx.beginPath();
    ctx.arc(x, y, r, 0, 2 * Math.PI);
    ctx.fillStyle = "#a371f7";
    ctx.fill();
    requestAnimationFrame(frame);
  }
  frame();
`} width={600} height={300} />

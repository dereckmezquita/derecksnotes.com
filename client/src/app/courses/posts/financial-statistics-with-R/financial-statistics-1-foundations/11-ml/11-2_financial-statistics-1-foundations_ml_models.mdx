## 11.4 Regression Models

### 11.4.1 Prose & Intuition

Linear regression forms the foundation of quantitative finance. Despite its simplicity, it remains surprisingly competitive with complex models when properly regularised and cross-validated.

The key insight: in low signal-to-noise environments, simpler models often outperform complex ones out-of-sample. Complex models overfit the noise that dominates financial data.

**Ridge regression** ($L_2$ penalty) shrinks coefficients toward zero, reducing variance at the cost of some bias. It works well when many features have small effects.

**LASSO** ($L_1$ penalty) drives some coefficients exactly to zero, performing feature selection. It's useful when we believe only a few features matter.

**Elastic Net** combines both penalties, offering a middle ground.

``` r
# Implement ridge regression manually
ridge_regression <- function(X, y, lambda) {
    # Add intercept
    X_int <- cbind(1, X)

    # Ridge solution: (X'X + lambda*I)^(-1) X'y
    XtX <- t(X_int) %*% X_int
    I <- diag(ncol(X_int))
    I[1, 1] <- 0  # Don't penalise intercept

    beta <- solve(XtX + lambda * I) %*% t(X_int) %*% y
    return(as.vector(beta))
}
```

``` r
# Implement coordinate descent LASSO (simplified)
soft_threshold <- function(x, lambda) {
    sign(x) * pmax(abs(x) - lambda, 0)
}

lasso_regression <- function(X, y, lambda, max_iter = 1000, tol = 1e-6) {
    n <- nrow(X)
    p <- ncol(X)

    # Standardise
    X_mean <- colMeans(X)
    X_sd <- apply(X, 2, sd)
    X_std <- scale(X)
    y_mean <- mean(y)
    y_std <- y - y_mean

    # Initialise
    beta <- rep(0, p)

    for (iter in 1:max_iter) {
        beta_old <- beta

        for (j in 1:p) {
            # Partial residual
            r <- y_std - X_std[, -j, drop = FALSE] %*% beta[-j]

            # Coordinate update
            rho <- sum(X_std[, j] * r) / n
            beta[j] <- soft_threshold(rho, lambda)
        }

        if (max(abs(beta - beta_old)) < tol) break
    }

    # Transform back
    beta_orig <- beta / X_sd
    intercept <- y_mean - sum(beta_orig * X_mean)

    return(list(intercept = intercept, beta = beta_orig))
}

# Calculate LASSO paths
lambdas_lasso <- seq(0.0001, 0.005, 0.0002)

        lambda = lam,
```

``` r
# Compare OLS, Ridge, and LASSO via walk-forward CV
compare_models <- function(train, test, feature_cols, target_col) {
    X_train <- as.matrix(train[, ..feature_cols])
    y_train <- train[[target_col]]
    X_test <- as.matrix(test[, ..feature_cols])
    y_test <- test[[target_col]]

    results <- list()

    # OLS
    ols_fit <- lm(as.formula(paste(target_col, "~", paste(feature_cols, collapse = "+"))), data = train)
    ols_pred <- predict(ols_fit, test)
    results$ols_ic <- cor(ols_pred, y_test, use = "complete.obs")

    # Ridge (fixed lambda)
    ridge_coef <- ridge_regression(X_train, y_train, lambda = 10)
    ridge_pred <- cbind(1, X_test) %*% ridge_coef
    results$ridge_ic <- cor(ridge_pred, y_test, use = "complete.obs")

    # LASSO (fixed lambda)
    lasso_fit <- lasso_regression(X_train, y_train, lambda = 0.001)
    lasso_pred <- lasso_fit$intercept + X_test %*% lasso_fit$beta
    results$lasso_ic <- cor(lasso_pred, y_test, use = "complete.obs")

    return(results)
}
```

### 11.4.3 Mathematical Derivation

**Ordinary Least Squares**

$$
\hat{\beta}_{OLS} = \arg\min_\beta \|y - X\beta\|_2^2 = (X'X)^{-1}X'y
$$

**Ridge Regression**

$$
\hat{\beta}_{Ridge} = \arg\min_\beta \|y - X\beta\|_2^2 + \lambda\|\beta\|_2^2 = (X'X + \lambda I)^{-1}X'y
$$

**LASSO**

$$
\hat{\beta}_{LASSO} = \arg\min_\beta \|y - X\beta\|_2^2 + \lambda\|\beta\|_1
$$

No closed-form solution; solved via coordinate descent.

**Elastic Net**

$$
\hat{\beta}_{EN} = \arg\min_\beta \|y - X\beta\|_2^2 + \lambda_1\|\beta\|_1 + \lambda_2\|\beta\|_2^2
$$

**Bias-Variance Decomposition**

For Ridge: as $\lambda \to 0$, $\hat{\beta} \to \hat{\beta}_{OLS}$ (high variance, low bias)

As $\lambda \to \infty$, $\hat{\beta} \to 0$ (low variance, high bias)

Optimal $\lambda$ balances this tradeoff.

``` r
# Find optimal lambda via inner CV
find_optimal_lambda <- function(X, y, lambdas, n_folds = 5) {
    n <- length(y)
    fold_size <- floor(n / n_folds)

    cv_errors <- numeric(length(lambdas))

    for (i in seq_along(lambdas)) {
        fold_errors <- numeric(n_folds)

        for (fold in 1:n_folds) {
            # Use time-ordered folds
            test_idx <- ((fold - 1) * fold_size + 1):(min(fold * fold_size, n))
            train_idx <- setdiff(1:n, test_idx)

            X_train <- X[train_idx, ]
            y_train <- y[train_idx]
            X_test <- X[test_idx, ]
            y_test <- y[test_idx]

            # Ridge fit
            coef <- ridge_regression(X_train, y_train, lambdas[i])
            pred <- cbind(1, X_test) %*% coef

            fold_errors[fold] <- mean((y_test - pred)^2)
        }

        cv_errors[i] <- mean(fold_errors)
    }

    return(data.table(lambda = lambdas, cv_error = cv_errors))
}

# Find optimal lambda
lambdas_cv <- 10^seq(-2, 3, 0.25)
cv_lambda <- find_optimal_lambda(X, y, lambdas_cv)

optimal_lambda <- cv_lambda[which.min(cv_error), lambda]

    geom_point(colour = tc[1], size = 2) +
    geom_vline(xintercept = log10(optimal_lambda), colour = tc[2], linetype = "dashed") +
```

## 11.5 Classification Models

### 11.5.1 Prose & Intuition

In finance, we often care about the **direction** of returns rather than their magnitude. Classifying whether the market will go up or down tomorrow is often more tractable than predicting the exact return.

**Logistic regression** is the natural extension of linear regression to classification. It models the probability of an outcome, making it useful for position sizing.

**Decision trees** partition the feature space into regions with different predictions. They're interpretable but prone to overfitting.

**Random forests** and **gradient boosting** combine many trees to improve prediction. They're powerful but require careful regularisation in financial applications.

The key challenge: classification accuracy is a poor metric in finance. A model that correctly predicts 90% of small moves but misses all large moves is worthless.

``` r
# Implement logistic regression via gradient descent
logistic_regression <- function(X, y, lr = 0.01, max_iter = 1000, tol = 1e-6) {
    n <- nrow(X)
    X_int <- cbind(1, X)  # Add intercept
    p <- ncol(X_int)

    beta <- rep(0, p)

    sigmoid <- function(z) 1 / (1 + exp(-z))

    for (iter in 1:max_iter) {
        # Predictions
        z <- X_int %*% beta
        p_hat <- sigmoid(z)

        # Gradient
        grad <- t(X_int) %*% (p_hat - y) / n

        # Update
        beta_new <- beta - lr * grad

        if (max(abs(beta_new - beta)) < tol) break
        beta <- beta_new
    }

    return(beta)
}
```

### 11.5.3 Mathematical Derivation

**Logistic Regression**

$$
P(y=1|x) = \sigma(\beta'x) = \frac{1}{1 + e^{-\beta'x}}
$$

**Log-Likelihood**

$$
\mathcal{L}(\beta) = \sum_{i=1}^{n} \left[ y_i \log(\hat{p}_i) + (1-y_i)\log(1-\hat{p}_i) \right]
$$

**Gradient**

$$
\nabla_\beta \mathcal{L} = X'(y - \hat{p})
$$

**Decision Threshold**

Default: predict 1 if $\hat{p} > 0.5$

Optimal threshold: minimise expected loss given asymmetric costs

$$
\text{threshold} = \frac{c_{FP}}{c_{FP} + c_{FN}}
$$

where $c_{FP}$ and $c_{FN}$ are false positive and false negative costs.

**Trading Application**

Position size proportional to edge:

$$
\text{position} = \frac{2\hat{p} - 1}{\sigma}
$$

where $\sigma$ is position volatility target.

## 11.6 Tree-Based Models

### 11.6.1 Prose & Intuition

Decision trees recursively partition the feature space by finding optimal split points. They're highly interpretable—you can trace exactly why a prediction was made.

However, single trees are unstable and prone to overfitting. Small changes in data can lead to completely different trees.

**Random Forests** address this by training many trees on bootstrap samples with random feature subsets. The ensemble prediction averages out individual tree errors.

**Gradient Boosting** builds trees sequentially, with each tree correcting errors from previous trees. It's powerful but requires careful regularisation.

In finance, tree-based models can capture non-linear relationships and interactions that linear models miss. But they're also more prone to overfitting spurious patterns.

### 11.6.2 Visual Evidence


``` r
# Implement simple decision tree (binary splits only)
find_best_split <- function(X, y, feature_idx, min_samples = 20) {
    x <- X[, feature_idx]
    n <- length(y)

    if (n < min_samples * 2) return(list(gain = -Inf))

    # Try all unique values as split points
    unique_vals <- sort(unique(x))
    if (length(unique_vals) < 2) return(list(gain = -Inf))

    best_gain <- -Inf
    best_threshold <- NA

    var_total <- var(y) * n

    for (i in 1:(length(unique_vals) - 1)) {
        threshold <- (unique_vals[i] + unique_vals[i + 1]) / 2

        left_idx <- x <= threshold
        right_idx <- x > threshold

        n_left <- sum(left_idx)
        n_right <- sum(right_idx)

        if (n_left < min_samples || n_right < min_samples) next

        var_left <- var(y[left_idx]) * n_left
        var_right <- var(y[right_idx]) * n_right

        gain <- var_total - var_left - var_right

        if (gain > best_gain) {
            best_gain <- gain
            best_threshold <- threshold
        }
    }

    return(list(gain = best_gain, threshold = best_threshold, feature = feature_idx))
}

# Train simple tree (depth 2)
tree_data <- data.table(X = X, y = y)
```

``` r
# Implement simplified random forest
random_forest_predict <- function(X, y, X_test, n_trees = 50, max_depth = 3, mtry = 3) {

    n <- nrow(X)
    predictions <- matrix(0, nrow = nrow(X_test), ncol = n_trees)

    for (tree in 1:n_trees) {
        # Bootstrap sample
        boot_idx <- sample(1:n, n, replace = TRUE)
        X_boot <- X[boot_idx, ]
        y_boot <- y[boot_idx]

        # Random feature subset
        features <- sample(1:ncol(X), mtry)

        # Simple: use mean of bootstrap for prediction (placeholder)
        # In practice, build full tree
        pred <- mean(y_boot)

        # Add noise based on feature importance (simulated)
        for (f in features) {
            # Correlation-based adjustment
            cor_f <- cor(X_boot[, f], y_boot, use = "complete.obs")
            pred <- pred + cor_f * X_test[, f] * 0.1
        }

        predictions[, tree] <- pred
    }

    return(rowMeans(predictions))
}
```

### 11.6.3 Mathematical Derivation

**Decision Tree Split Criterion**

For regression (variance reduction):

$$
\text{Gain} = Var(y) - \frac{n_L}{n}Var(y_L) - \frac{n_R}{n}Var(y_R)
$$

For classification (Gini impurity):

$$
G = 1 - \sum_{k=1}^{K} p_k^2
$$

**Random Forest**

Prediction: $\hat{f}(x) = \frac{1}{B}\sum_{b=1}^{B} T_b(x)$

where $T_b$ is tree $b$ trained on bootstrap sample with random features.

**Gradient Boosting**

Sequential update: $F_m(x) = F_{m-1}(x) + \gamma_m h_m(x)$

where $h_m$ is fitted to negative gradient (pseudo-residuals):

$$
r_{im} = -\frac{\partial L(y_i, F(x_i))}{\partial F(x_i)} \Big|_{F=F_{m-1}}
$$

**Regularisation**

Learning rate: $F_m(x) = F_{m-1}(x) + \eta \cdot h_m(x)$

Tree complexity: limit max depth, min samples per leaf

Subsampling: train on random subset of data each iteration

``` r
ensemble_results <- list()
start_idx <- train_window + 1

while (start_idx + test_window <= nrow(data)) {
    train_idx <- (start_idx - train_window):(start_idx - 1)
    test_idx <- start_idx:(start_idx + test_window - 1)

    train <- data[train_idx]
    test <- data[test_idx]

    # Model 1: Momentum features
    features1 <- c("lag1", "mom_20", "mom_60")
    X1_train <- scale(as.matrix(train[, ..features1]))
    X1_test <- scale(as.matrix(test[, ..features1]),
                    center = attr(X1_train, "scaled:center"),
                    scale = attr(X1_train, "scaled:scale"))
    coef1 <- ridge_regression(X1_train, train$target, lambda = 10)
    pred1 <- cbind(1, X1_test) %*% coef1

    # Model 2: Factor features
    features2 <- c("mkt_lag", "smb_lag", "hml_lag")
    X2_train <- scale(as.matrix(train[, ..features2]))
    X2_test <- scale(as.matrix(test[, ..features2]),
                    center = attr(X2_train, "scaled:center"),
                    scale = attr(X2_train, "scaled:scale"))
    coef2 <- ridge_regression(X2_train, train$target, lambda = 10)
    pred2 <- cbind(1, X2_test) %*% coef2

    # Model 3: Volatility features
    features3 <- c("vol_20", "ma_ratio")
    X3_train <- scale(as.matrix(train[, ..features3]))
    X3_test <- scale(as.matrix(test[, ..features3]),
                    center = attr(X3_train, "scaled:center"),
                    scale = attr(X3_train, "scaled:scale"))
    coef3 <- ridge_regression(X3_train, train$target, lambda = 10)
    pred3 <- cbind(1, X3_test) %*% coef3

    # Ensemble: simple average
    pred_ensemble <- (pred1 + pred2 + pred3) / 3

    # Calculate ICs
    y_test <- test$target
    ensemble_results[[length(ensemble_results) + 1]] <- data.table(
        fold = length(ensemble_results) + 1,
        test_start = data[start_idx, date],

        ic_vol = cor(pred3, y_test, use = "complete.obs"),
        ic_ensemble = cor(pred_ensemble, y_test, use = "complete.obs")
    )

    start_idx <- start_idx + step_size
}
```
